#/bin/bash

##this script will be executed every five readings
##to find avg of values in memory file
echo "Memory Used Average at TimeStmamp `date`  " >>/var/www/html/memoryAvg 
cat /var/www/html/memory| awk '{ memtotal += $1; count1++ } END { print memtotal/count1  }' >> /var/www/html/memoryAvg
echo -n "" >/var/www/html/memory ## to clean the current five readings 

##to find avg of values in cpu idle file
echo "CPU Idle Average at TimeStmamp `date`  " >>/var/www/html/cpuAvg
cat /var/www/html/cpu| awk '{ cpuitotal += $1; count2++ } END { print cpuitotal/count2  }' >> /var/www/html/cpuAvg
echo -n "" >/var/www/html/cpu ## to clean the current five readings

##to find avg of values in cpu usedfile
echo "CPU Used Average:\n" >>/var/www/html/cpuAvg 
cat /var/www/html/cpuused| awk '{ cpuutotal += $1; count3++ } END { print cpuutotal/count3  }' >> /var/www/html/cpuAvg
echo -n "" >/var/www/html/cpuused ## to clean the current five readings

##to find avg of values in doskused file
echo "Disk Used Avg at TimeStmamp `date`  " >>/var/www/html/diskAvg
cat /var/www/html/disk| awk '{ diskuutotal += $1; count5++ } END { print diskuutotal/count5  }' >> /var/www/html/diskAvg 
echo -n "" >/var/www/html/disk ## to clean the current five readings

##to find avg of values in disk free file
echo "Disk Free at TimeStamp `date` " >>/var/www/html/diskAvg
cat /var/www/html/diskfree| awk '{ diskfftotal += $1; count7++ } END { print diskfftotal/count7  }' >> /var/www/html/diskAvg 
echo -n "" >/var/www/html/diskfree ## to clean the current five readings
